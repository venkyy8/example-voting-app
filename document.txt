git clone https://github.com/dockersamples/example-voting-app.git
cd example-voting-app

docker network create voting-app

#redis
docker run -d \
  --name redis \
  --network voting-app \
  redis:alpine

#postgreSQL
docker run -d \
  --name db \
  --network voting-app \
  -e POSTGRES_USER=postgres \
  -e POSTGRES_PASSWORD=postgres \
  postgres:15-alpine

#vote# python
cd vote

docker build -t voting-app-vote .

docker run -d \
  --name vote \
  --network voting-app \
  -p 5000:80 \
  voting-app-vote

#results # Nodejs
cd result

docker build -t voting-app-result .

docker run -d \
  --name result \
  --network voting-app \
  -p 5001:80 \
  voting-app-result

#worker # .net
cd worker

docker build -t voting-app-worker .

docker run -d \
  --name worker \
  --network voting-app \
  voting-app-worker
